
	List
		[Get]
			InitView();
			PrepareForList(model); --> * DecodeModelBeforeView(item1);
	
	Add
		[Get]
			InitView(model); --> InitView();
			PrepareForAdd(model); --> Prepare(model);
		[Post]
			FixModelAfterInput(model);
			BeforeAdd(model); => BeforeChange(model);
			EncodeModelBeforeSave(model);
			ValidationModel(model);			
			<Add>
			AfterAdd(model); => AfterChange(model);
	
	Details
		[Get]
			DecodeModelBeforeView(model1);
			InitView(model); --> InitView();
			PrepareForDetails(model); --> Prepare(model);
	
	Edit
		[Get]
			DecodeModelBeforeEdit(model1);
			InitView(model); --> InitView();
			PrepareForEdit(model); --> PrepareForDetails(model);
		[Post]
			FixModelAfterInput(model);
			BeforeUpdate(model); => BeforeChange(model);
			EncodeModelBeforeSave(model);
			ValidationModel(model);			
			<Update>
			AfterUpdate(model); => AfterChange(model);

	Delete
		[Get]
			DecodeModelBeforeView(model1);
			InitView(model); --> InitView();
			PrepareForDelete(model); --> PrepareForDetails(model);
		[Post]
			BeforeDelete(model1);
			<Remove>
			AfterDelete();
